\newcommand{\professionalexperience}{
    \section{Professional Experience}
    
    % Current/Recent Position Template
    \cventry{[START_YEAR]--Present}{[POSITION_TITLE]}{[COMPANY_NAME]}{[LOCATION]}{}
    {
    \begin{itemize}[leftmargin=*]
    \item [ACHIEVEMENT_WITH_METRICS] - Focus on quantifiable results
    \item [TECHNICAL_IMPLEMENTATION] - Highlight technical skills used
    \item [PROCESS_IMPROVEMENT] - Show impact on efficiency/quality
    \item [COLLABORATION] - Demonstrate teamwork and leadership
    \item [INNOVATION] - Showcase problem-solving and innovation
    \end{itemize}
    }

    \cventry{[START_YEAR]--[END_YEAR]}{[POSITION_TITLE]}{[COMPANY_NAME]}{[LOCATION]}{}

    \item [COLLABORATION]
    \end{itemize}
    }
}
